https://leetcode.com/problems/two-sum-ii-input-array-is-sorted/

O(n) S(1)

class Solution {
    public int[] twoSum(int[] numbers, int target) {
        int n = numbers.length;
        int li = 0;
        int ri = n-1;
        while(li < ri){
            int leftVal = numbers[li];
            int rightVal = numbers[ri];
            int currSum = leftVal + rightVal;
            if(currSum == target){
                return new int[]{li+1,ri+1};
            }else if( currSum < target ) li++;
            else ri--;
        }
         return new int[]{0,0};
    }
}
