O(n) S(1)

class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int minutes) {
        int i1 = 0;
        int i2 = 0;
        int maxSatisfiedModified = 0;
        int satisfiedModified = 0;
        for(int i = 0;i<minutes;i++){
            if(grumpy[i] == 1 ) satisfiedModified += customers[i];
        }
        maxSatisfiedModified = satisfiedModified;
        i1 = 0; i2=minutes-1;
        for(int i = minutes; i<customers.length;i++){
            if(grumpy[i - minutes] == 1) satisfiedModified -= customers[i - minutes ];
            if(grumpy[i] == 1) satisfiedModified += customers[i];
            if( maxSatisfiedModified < satisfiedModified){
                maxSatisfiedModified = satisfiedModified;
                i1 = i - minutes + 1;
                i2 = i;
            }
        }
        int res = 0;
        for(int i = 0;i<customers.length;i++){
            if(grumpy[i] == 0 || (i >= i1 && i <= i2 )) res+=customers[i];
        }
        return res;
    }
}
