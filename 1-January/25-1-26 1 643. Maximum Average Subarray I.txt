https://leetcode.com/problems/maximum-average-subarray-i/

O(n) S(1)

class Solution {
    public double findMaxAverage(int[] nums, int k) {
        double currSum = 0;
        double maxAvg = 0;
        for(int i = 0;i<k;i++){
            currSum+=nums[i];
        }
        maxAvg = currSum / k;
        for(int i = k;i<nums.length;i++){
            currSum = currSum - nums[i-k] + nums[i];
            maxAvg = Math.max(currSum/k, maxAvg);
        }
        return maxAvg;
    }
}
