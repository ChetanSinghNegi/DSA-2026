https://leetcode.com/problems/3sum/

O(n2) S(1)

class Solution {

    public List<List<Integer>> threeSum(int[] nums) {
        Arrays.sort(nums);
        List<List<Integer>> res = new ArrayList<>();
        int n = nums.length;

        for (int i = 0; i < n - 2; i++) {

            // skip duplicate fixed elements
            if (i > 0 && nums[i] == nums[i - 1]) continue;

            twoSum(nums, i + 1, -nums[i], res);
        }

        return res;
    }

    private void twoSum(int[] nums, int leftIdx, int target, List<List<Integer>> res) {
        int rightIdx = nums.length - 1;

        while (leftIdx < rightIdx) {
            int sum = nums[leftIdx] + nums[rightIdx];

            if (sum == target) {
                res.add(Arrays.asList(-1*target, nums[leftIdx], nums[rightIdx]));

                leftIdx++;
                rightIdx--;

                // skip duplicates for left and right
                while (leftIdx < rightIdx && nums[leftIdx] == nums[leftIdx - 1]) leftIdx++;
                while (leftIdx < rightIdx && nums[rightIdx] == nums[rightIdx + 1]) rightIdx--;

            } else if (sum < target) {
                leftIdx++;
            } else {
                rightIdx--;
            }
        }
    }
}
