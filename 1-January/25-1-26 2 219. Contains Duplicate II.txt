https://leetcode.com/problems/contains-duplicate-ii/

O(n) S(1)

class Solution {
    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashMap<Integer, Integer> hm = new HashMap<>(); //number, index
        for(int i = 0;i<nums.length;i++){
            int ele = nums[i];
            int idx = hm.getOrDefault(ele,-1);
            if(idx != -1 && Math.abs(idx - i)<=k) return true;
            hm.put(ele,i);
        }
        return false;
    }
}
