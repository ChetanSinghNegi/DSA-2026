O(rowIndex) S(rowIndex)

class Solution {
    public List<Integer> getRow(int rowIndex) {
        List<Integer> ans = new ArrayList<>();
        ans.add(1);
        long prev = 1;
        for(int k = 1;k<=rowIndex;k++){
            long currVal = prev*(rowIndex - k + 1) / k;
            ans.add((int)currVal);
            prev = currVal; 
        }
        return ans;
    }
}
