O(n) S(n)

class NumArray {
    int[] prefixSumArr;
    public NumArray(int[] nums) {
        int n =nums.length;
        prefixSumArr = new int[n];
        int currSum = 0;
        for(int i = 0;i<n;i++){
            if(i ==0){
                currSum = nums[i];
                prefixSumArr[i] = currSum;
                continue;
            }
            currSum += nums[i];
            prefixSumArr[i] = currSum;
        }
    }
    
    public int sumRange(int left, int right) {
        int tillPrefixSum = left - 1 >= 0 ? prefixSumArr[left - 1] : 0;
        int res = prefixSumArr[right] - tillPrefixSum;
        return res;
    }
}

/**
 * Your NumArray object will be instantiated and called as such:
 * NumArray obj = new NumArray(nums);
 * int param_1 = obj.sumRange(left,right);
 */
