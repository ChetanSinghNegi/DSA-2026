https://leetcode.com/problems/maximum-product-subarray/
O(n) S(1)

class Solution {
    public int maxProduct(int[] nums) {
        int pre = 1, suf = 1, maxMul = Integer.MIN_VALUE, n = nums.length;
        for(int i = 0;i<n;i++){
            if(pre == 0) pre = 1;
            if(suf == 0) suf = 1;
            pre = pre * nums[i];
            suf = suf * nums[n-i-1];
            maxMul = Math.max(maxMul, Math.max(pre, suf));
        }

        return maxMul;        
    }
}
