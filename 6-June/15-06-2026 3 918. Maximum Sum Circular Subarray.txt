https://leetcode.com/problems/maximum-sum-circular-subarray/
O(n) S(1)

class Solution {
    public int maxSubarraySumCircular(int[] nums) {
        int currMaxSum = nums[0], maxSum = nums[0], currMinSum = nums[0], minSum=nums[0], totalSum = nums[0];
        for(int i = 1;i<nums.length;i++){
            currMaxSum = Math.max(nums[i], currMaxSum+nums[i]);
            maxSum = Math.max(maxSum, currMaxSum);

            currMinSum = Math.min(nums[i], currMinSum + nums[i]);
            minSum = Math.min(minSum, currMinSum);

            totalSum += nums[i];
        }
        if(totalSum == minSum) return maxSum;
        return Math.max(maxSum, totalSum - minSum); //totalArray - MinSubarray = MaxSubarray
    }
}
