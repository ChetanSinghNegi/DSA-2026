https://leetcode.com/problems/car-fleet/
O(nlogn) S(n)

class Solution {
    class Pair {
        int pos;
        double time;

        Pair(int pos, double time) {
            this.pos = pos;
            this.time = time;
        }
    }

    public int carFleet(int target, int[] position, int[] speed) {

        int n = position.length;
        List<Pair> al = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            double time = (double) (target - position[i]) / speed[i];
            al.add(new Pair(position[i], time));
        }

        // Sort by position
        Collections.sort(al, (a, b) -> Double.compare(a.pos, b.pos));

        int fleets = 0;
        double maxTime = 0;

        // Traverse from the car nearest the target
        for (int i = n - 1; i >= 0; i--) {
            if (al.get(i).time > maxTime) {
                fleets++;
                maxTime = al.get(i).time;
            }
        }

        return fleets;
    }
}
