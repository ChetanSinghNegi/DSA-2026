O(h) S(rec(h) + aux(1))
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isSymmetric(TreeNode root) {
        return helper(root, root);
    }
    public boolean helper(TreeNode leftNode, TreeNode rightNode){
        if(leftNode == null && rightNode == null) return true;
        else if(leftNode == null || rightNode == null) return false;
        else if(leftNode.val != rightNode.val) return false;
        boolean isLeftTreeSymmetric = helper(leftNode.left, rightNode.right);
        if(isLeftTreeSymmetric == false) return false;
        return helper(leftNode.right, rightNode.left);
    }
}
