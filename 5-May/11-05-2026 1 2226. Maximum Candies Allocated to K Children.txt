https://leetcode.com/problems/maximum-candies-allocated-to-k-children/
O(nlog(maxEle)) S(1)

class Solution {
    public int maximumCandies(int[] candies, long k) {
        int minCandies = 1, maxCandies = 1, finalCandies = 0;
        for(int candy:candies) maxCandies = Math.max(maxCandies, candy);
        while(minCandies <= maxCandies){
            long currCandies = (long)minCandies + (maxCandies - minCandies)/2;
            boolean validCandiesFlag = isValidCandies(currCandies, k, candies);
            if(validCandiesFlag == true){
                finalCandies = (int)currCandies;
                minCandies = (int)currCandies+1;
            }else{
                maxCandies = (int)currCandies-1;
            }
        }
        return finalCandies;
    }
    public boolean isValidCandies(long currCandies, long k, int[] candies){
        long children = 0;
        for(int candy: candies){
            long currChildren = (long)candy/currCandies;
            children += currChildren;
            if(children >= k) return true;
        }
        return false;
    }
}
