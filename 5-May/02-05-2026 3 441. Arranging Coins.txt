https://leetcode.com/problems/arranging-coins/
O(logn) S(1)

class Solution {
    public int arrangeCoins(int n) {
        long nLong = (long)n, left = 0, right = nLong;
        int ans = 0;
        while(left <= right){
            long mid = left + (right - left)/2;
            long currEle = (mid) * (mid+1)/2;
            if(currEle <= n){
                ans = (int)mid;
                left = mid+1;
            }else right = mid-1;
        }
        return ans;
        
    }
}
