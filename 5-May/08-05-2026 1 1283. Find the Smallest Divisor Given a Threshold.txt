https://leetcode.com/problems/find-the-smallest-divisor-given-a-threshold/
O(nlog(maxElement)) S(1)

class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int min = 1, max = 1, ans = 0;
        for(int num:nums){
            max = Math.max(max, num);
        }

        while(min<=max){
            int mid = min + (max-min)/2;
            boolean isValid = isValidDivisor(mid, threshold, nums);
            if(isValid){
                ans = mid;
                max = mid-1;
            }else{
                min = mid+1;
            }
        }
        return ans;
    }
    public boolean isValidDivisor(int divisor, int threshold, int[] nums){
        int currThreshold = 0;
        for(int num:nums){
            int quotient = (num/divisor) + (num%divisor == 0 ? 0 : 1);
            currThreshold += quotient;
            if(currThreshold > threshold) return false;
        }
        return true;
    }
}
