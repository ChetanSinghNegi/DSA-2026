class Solution {
    public Node connect(Node root) {
        if(root == null) return root;
        int count = 1;
        Queue<Node> q = new ArrayDeque<>();
        q.add(root);
        while(q.size() > 0){
            while(count != 0){
                count--;
                Node remNode = q.remove();
                if(remNode.left != null ) q.add(remNode.left);
                if(remNode.right != null ) q.add(remNode.right); 
                if(count!=0){
                    Node peekNode = q.size()>0 ? q.peek() : null;
                    remNode.next = peekNode;
                }
            }
            count = q.size();
        }
        return root;
    }
}
