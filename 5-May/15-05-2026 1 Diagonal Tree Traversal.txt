M1) Iterative + Queue  O(n) S(n)

/* Node is defined as
class Node{
    int data;
    Node left;
    Node right;
    Node(int data){
        this.data = data;
        left=null;
        right=null;
    }
}*/
class Tree
{
     public ArrayList<Integer> diagonal(Node root)
      {
           //add your code here.
           ArrayList<Integer> ans=new ArrayList<>();
           Queue<Node> q=new ArrayDeque<>();
           q.add(root);
           Node curr;
           while(q.size()>0){
               curr=q.remove();
               while(curr!=null){
                   ans.add(curr.data);
                   if(curr.left!=null) q.add(curr.left);
                   curr=curr.right;
               }
           }
           return ans;
      }
}

M2) Recursive + List<List<Integer>>  O(n) S(rec(n) + aux(1))
/* Node is defined as
class Node{
    int data;
    Node left;
    Node right;
    Node(int data){
        this.data = data;
        left=null;
        right=null;
    }
}*/
class Tree {
    public ArrayList<Integer> diagonal(Node root) {
        ArrayList<ArrayList<Integer>> multiLevelElements = new ArrayList<>();
        helper(root, 0, multiLevelElements);
        ArrayList<Integer> ans = new ArrayList<>();
        for(ArrayList<Integer> levelElements: multiLevelElements){
            for(int ele: levelElements) ans.add(ele);
        }
        return ans;
        
    }
    public void helper(Node root, int level, ArrayList<ArrayList<Integer>> ans){
        if(root == null) return;
        if(level >= ans.size()) ans.add(new ArrayList<>());
        ArrayList<Integer> currAl = ans.get(level);
        currAl.add(root.data);
        helper(root.left, level+1, ans);
        helper(root.right, level, ans);
    }
}
