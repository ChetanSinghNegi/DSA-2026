https://leetcode.com/problems/path-sum/
O(n) S( Rec(height) Aux(1) )
class Solution {
    public boolean hasPathSum(TreeNode root, int targetSum) {
        return helper(root, 0, targetSum);
    }
    public boolean helper(TreeNode root, int sum, int targetSum){
        if(root == null) return false;
        sum += root.val;
        if(root.left == null && root.right == null) return sum == targetSum;
        boolean foundLeftTree = helper(root.left, sum, targetSum);
        if(foundLeftTree == true) return true;
        return helper(root.right, sum, targetSum);
    }
}
