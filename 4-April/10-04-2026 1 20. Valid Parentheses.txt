https://leetcode.com/problems/valid-parentheses/description/
O(n) S(n)

class Solution {
    public boolean isValid(String s) {
        Stack<Character> st = new Stack<>();
        for(char ch: s.toCharArray()){
            if(ch == '(' || ch == '{' || ch == '[') st.push(ch);
            else{
                if(st.size() == 0 || !isValidClosingCharacter(ch, st.pop())) return false;
            }
        }
        if(st.size() != 0) return false;
        return true;
    }
    
    public boolean isValidClosingCharacter(char ch, char topEle){
        if(ch == ')'){
            if(topEle != '(') return false;
        }else if(ch == '}'){
            if(topEle != '{') return false;
        }else if(ch == ']'){
            if(topEle != '[') return false;
        }
        return true;
    }
}
