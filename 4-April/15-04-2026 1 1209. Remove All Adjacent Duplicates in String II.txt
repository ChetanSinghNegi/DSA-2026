https://leetcode.com/problems/remove-all-adjacent-duplicates-in-string-ii/
O(n) S(n)

class Solution {
    public static class Pair{
        char pairChar;
        int rep;
        Pair(char pairChar, int rep){
            this.pairChar = pairChar;
            this.rep = rep;
        }
    }
    public String removeDuplicates(String s, int k) {
        Stack<Pair> stk = new Stack<>();
        for (char ch: s.toCharArray()){
            if(stk.size() == 0){
                stk.push(new Pair(ch, 1));
                continue;
            }
            Pair topPair = stk.peek();
            if(topPair.pairChar == ch){
                topPair.rep++;
                if(topPair.rep == k){
                    stk.pop();
                }
            }else{
                stk.push(new Pair(ch, 1));
            }
        }
        StringBuilder sb = new StringBuilder("");
        while(stk.size() > 0){
            Pair pair = stk.pop();
            for(int i = 0;i<pair.rep;i++){
                sb.append(pair.pairChar);
            }
        }
        return sb.reverse().toString();
    }
}
