/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode[] splitListToParts(ListNode head, int k) {
        ListNode[] answer = new ListNode[k];
        if(head == null) return answer;
        int sz = getLLSize(head); 
        int kClone = k;
        for(int i = 0; i< k && head!=null;i++){
            int currChildLLSize = sz/kClone;
            if(sz % kClone != 0) currChildLLSize++;
            ListNode newHead = getSubLL(head, currChildLLSize);
            answer[i] = head;
            head = newHead;
            sz -= currChildLLSize;
            kClone--;
        }
        return answer;
    }
    public ListNode getSubLL(ListNode head, int size){
        if(head == null) return null;
        while(size != 1 && head != null){
            head = head.next;
            size--;
        }
        if(head == null) return null;
        ListNode headNext = head.next;
        head.next = null;
        return headNext;
    }
    public int getLLSize(ListNode head){
        int sz = 0;
        while(head != null){
            sz++;
            head = head.next;
        }
        return sz;
    }
}
