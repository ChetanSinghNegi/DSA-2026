O(n2) S(1)

class Solution {
    public int triangleNumber(int[] nums) {
        Arrays.sort(nums);
        int totalConfig = 0;
        for(int i = nums.length-1;i>=2;i--){
            int largestSide = nums[i];
            int currValidConfig = findPair(i- 1, nums, largestSide);
            totalConfig += currValidConfig;
        }
        return totalConfig;
    }
    public int findPair(int right, int[] nums, int largestSide){
        int left = 0, totalPair = 0;
        while(left<right){
            int smallerSidesSum = nums[left]+nums[right];
            if(smallerSidesSum > largestSide){
                totalPair += right - left;
                right--;
            }else{
                left++;
            }
        }
        return totalPair;
    }
}
