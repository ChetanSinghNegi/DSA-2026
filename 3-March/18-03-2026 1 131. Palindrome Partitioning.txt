https://leetcode.com/problems/palindrome-partitioning/description/

TimeO(n · 2ⁿ)2ⁿ recursion nodes × O(n) palindrome check + copy per node

Space (auxiliary)O(n)Call stack depth + current path list => S(1)


class Solution {
    public List<List<String>> partition(String s) {
        List<List<String>> results = new ArrayList<>();
        List<String> result = new ArrayList<>();
        helper(s, result, results);
        return results;
    }
    public boolean isPalindrome(String str){
        for(int i = 0; i< str.length()/2;i++){
            if(str.charAt(i) != str.charAt(str.length() - i - 1)) return false;
        }
        return true;
    }
    public void helper(String str, List<String> result, List<List<String>> results){
        if(str.length() == 0){
            results.add(new ArrayList<>(result));
            return;
        }
        for(int i = 0; i<str.length();i++){
            String substring1 = str.substring(0,i+1);
            String substring2 = i+1< str.length() ? str.substring(i+1) : "";
            boolean isPalindromicString = isPalindrome(substring1);
            if(isPalindromicString){
                result.add(substring1);
                helper(substring2, result, results);
                result.remove(result.size()-1);
            }
        }
    }
}
