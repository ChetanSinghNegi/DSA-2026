Space Complexity: S ( 𝑛 ⋅ 𝑛 ! ) (dominated by output storage; auxiliary space is only S ( 𝑛 ) )

import java.util.*;

class Solution {
    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        Arrays.sort(nums); // sort to handle duplicates
        boolean[] used = new boolean[nums.length];
        backtrack(nums, used, new ArrayList<>(), result);
        return result;
    }

    private void backtrack(int[] nums, boolean[] used, List<Integer> path, List<List<Integer>> result) {
        if (path.size() == nums.length) {
            result.add(new ArrayList<>(path));
            return;
        }

        for (int i = 0; i < nums.length; i++) {
            if (used[i]) continue;

            // Skip duplicates
            if (i > 0 && nums[i] == nums[i-1] && !used[i-1]) continue;

            used[i] = true;
            path.add(nums[i]);

            backtrack(nums, used, path, result);

            path.remove(path.size() - 1);
            used[i] = false;
        }
    }
}

class Solution {
    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> uniquePermutations = new ArrayList<>();
        List<Integer> uniquePermutation = new ArrayList<>();
        HashMap<Integer, Integer> freq = new HashMap<>();
        HashSet<Integer> uniqueNums = new HashSet<>();
        for(int ele:nums){
            uniqueNums.add(ele);
            freq.put(ele, freq.getOrDefault(ele,0)+1);
        }
        helper(freq, uniqueNums, uniquePermutation, uniquePermutations);
        return uniquePermutations;
    }

    public void helper( HashMap<Integer, Integer> freq, HashSet<Integer> uniqueNums, List<Integer> uniquePermutation, List<List<Integer>> uniquePermutations){
        if(freq.size()==0){
            List<Integer> base = new ArrayList<>(uniquePermutation);
            uniquePermutations.add(base);
            return;
        }
        for(int key:uniqueNums){
            if(!freq.containsKey(key)) continue;
            freq.put(key, freq.get(key)-1);
            if(freq.get(key)==0) freq.remove(key);
            uniquePermutation.add(key);
            helper(freq, uniqueNums, uniquePermutation, uniquePermutations);
            uniquePermutation.remove(uniquePermutation.size()-1);
            freq.put(key, freq.getOrDefault(key, 0)+1);
        }
    }
}
