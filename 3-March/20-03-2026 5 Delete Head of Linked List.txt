O(1) S(1)

/*
class Node
{
    int data;
    Node next;

    Node(int data)
    {
        this.data = data;
        this.next = next;
    }
}
*/

class Solution {
    public Node deleteHead(Node head) {
        if(head == null) return null;
        Node newHead = head.next;
        head.data = 0;
        head.next = null;
        return newHead;
    }
}
