https://leetcode.com/problems/combination-sum-ii/description/
O(2^n*k) 
S( rec-n + 1) ==> S(1)

class Solution {
    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        List<List<Integer>> combinations = new ArrayList<>();
        List<Integer> combination = new ArrayList<>();
        Arrays.sort(candidates);
        helper(0, 0, target, candidates, combination, combinations);
        return combinations;
    }
    public void helper(int idx, int currSum, int target, int[] candidates, List<Integer> combination, List<List<Integer>> combinations){
        if(currSum == target) {
            combinations.add(new ArrayList<>(combination));
            return;
        }
        if(currSum > target || idx == candidates.length) return;
        for(int i = idx;i<candidates.length;i++){
            if(i != idx && candidates[i] == candidates[i-1]) continue;
            combination.add(candidates[i]);
            helper(i+1, currSum + candidates[i], target, candidates, combination, combinations);
            combination.remove(combination.size()-1);
        }
    }
}
