https://leetcode.com/problems/get-equal-substrings-within-budget/

O(n) S(1)

class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        int st = 0, ed = 0, currSum = 0, maxCount = 0, n = s.length();

        while(ed < n){
            char ch1 = s.charAt(ed);
            char ch2 = t.charAt(ed);
            int diff = Math.abs(ch1-ch2);
            currSum += diff;
            while(st < n && currSum > maxCost){
                int prevDiff = Math.abs(s.charAt(st) - t.charAt(st));
                currSum -= prevDiff;
                st++;
            }
            maxCount = Math.max(maxCount, ed-st+1);
            ed++;
        }

        return maxCount;
    }
}
