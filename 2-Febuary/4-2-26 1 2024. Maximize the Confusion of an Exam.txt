O(n) S(1)
class Solution {
    public int maxConsecutiveAnswers(String answerKey, int k) {
        int n = answerKey.length();
        int maxSubarrayLen = 0;
        int stFalse = 0, edFalse = 0, flipsCompletedFalse = 0;
        int stTrue = 0, edTrue = 0, flipsCompletedTrue = 0;

        while(edFalse < n){
            // For Maximum Chain of True 
            char ch = answerKey.charAt(edFalse);  
            if(ch == 'F'){
                if(flipsCompletedFalse == k){
                    while(stFalse < n && answerKey.charAt(stFalse) == 'T'){
                        stFalse++;
                    }
                    stFalse++;
                }
                else{
                    flipsCompletedFalse++;
                }
            }
            maxSubarrayLen = Math.max(maxSubarrayLen, edFalse - stFalse + 1);
            edFalse++;

             // For Maximum Chain of False 
            if(ch == 'T'){
                if(flipsCompletedTrue == k){
                    while(stTrue < n && answerKey.charAt(stTrue) == 'F'){
                        stTrue++;
                    }
                    stTrue++;
                }
                else{
                    flipsCompletedTrue++;
                }
            }
            maxSubarrayLen = Math.max(maxSubarrayLen, edTrue - stTrue + 1);
            edTrue++;
        }

        return maxSubarrayLen;
    }
}

M2) O(n) S(1)
class Solution {
    public int maxConsecutiveAnswers(String answerKey, int k) {

        int left = 0;
        int n = answerKey.length();
        int maxLength = 0;

        int countT = 0;
        int countF = 0;

        for (int right = 0; right < n; right++) {
            if (answerKey.charAt(right) == 'T') {
                countT++;
            } else {
                countF++;
            }

            while (Math.min(countT, countF) > k) {
                if (answerKey.charAt(left) == 'T') {
                    countT--;
                } else {
                    countF--;
                }
                left++;
            }

            maxLength = Math.max(maxLength, right - left + 1);
        }

        return maxLength;
    }
}
