https://leetcode.com/problems/fibonacci-number/
O(n) S(n)

class Solution {
    
    public int fib(int n) {
        if( n == 0 || n == 1 ) return n;
        int[] dp = new int[n+1];
        dp[0] = 0; dp[1] = 1;
        return helper(n, dp);
    }

    public int helper(int n, int[] dp){
        if( n == 0 ) return 0;
        if(dp[n] != 0) return dp[n];
        int n1 = helper(n-1, dp);
        int n2 = helper(n-2, dp);
        dp[n] = n1+n2;
        return n1+n2;
    }
}
